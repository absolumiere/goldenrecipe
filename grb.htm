<script>

const clickSound = document.getElementById('click-sound');

const shoppingList = document.getElementById('shopping-list');

const levelDescription = document.getElementById('level-description');

// Removed newItemInput and addItemButton as the section is removed

 

// Recipe Modal Elements

const recipeModalOverlay = document.getElementById('recipe-modal-overlay');

const modalRecipeTitle = document.getElementById('modal-recipe-title');

const modalRecipeItems = document.getElementById('modal-recipe-items');

 

// Alert Modal Elements

const alertModalOverlay = document.getElementById('alert-modal-overlay');

const alertModalTitle = document.getElementById('alert-modal-title');

const alertModalMessage = document.getElementById('alert-modal-message');

 

const levels = [

{

desc: "I want something soft and buttery...",

items: ['Butter', 'Flour', 'Sugar']

},

{

desc: "I'm craving something juicy and tangy!",

items: ['Orange Juice', 'Lemon Juice', 'Water']

},

{

desc: "Maybe something perfect for breakfast?",

items: ['Eggs', 'Milk', 'Syrup']

},

{

desc: "Banana... sounds good today.",

items: ['Banana', 'Oil', 'Flour']

},

{

desc: "I want something refreshing and crunchy.",

items: ['Lettuce', 'Tomato', 'Dressing']

},

{

desc: "Classic snack vibes please.",

items: ['Butter', 'Vanilla', 'Flour']

}

];

let currentLevel = 0;

// Removed userAddedItems as the add section is removed

 

/**

* Loads the items and description for the current level into the checklist.

* @param {number} level The index of the level to load.

*/

function loadLevel(level) {

const data = levels[level];

// Add quotation marks to the description

levelDescription.textContent = `"${data.desc}"`;

 

renderShoppingList(); // Call render function to populate the list

}

 

/**

* Renders the shopping list combining predefined items.

*/

function renderShoppingList() {

const data = levels[currentLevel];

// Only use predefined items now

const allItems = [...data.items];

 

shoppingList.innerHTML = allItems.map(item =>

`<li><label class="flex items-center"><input type="checkbox" class="custom-checkbox" data-item="${item}"><span class="text-sm">${item}</span></label></li>`

).join('');

}

 

// Removed addItem function and event listeners for it

 

/**

* Changes the current level of the checklist.

* @param {number} direction The direction to change the level (-1 for previous, 1 for next).

*/

function changeLevel(direction) {

clickSound.play(); // Play sound on level change

currentLevel = (currentLevel + direction + levels.length) % levels.length;

loadLevel(currentLevel);

}

 

/**

* Displays the recipe details modal.

* @param {string} title The title of the recipe.

* @param {string[]} items An array of ingredients for the recipe.

*/

function showRecipeModal(title, items) {

clickSound.play(); // Play sound on recipe card click

modalRecipeTitle.textContent = title;

modalRecipeItems.innerHTML = items.map(i => `<li>${i}</li>`).join('');

recipeModalOverlay.classList.add('show');

}

 

/**

* Hides the recipe details modal.

*/

function hideRecipeModal() {

clickSound.play(); // Play sound on closing modal

recipeModalOverlay.classList.remove('show');

}

 

/**

* Displays the custom alert modal with the given message.

* @param {string} message The message to display in the alert.

*/

function showAlertModal(message) {

clickSound.play(); // Play sound on showing alert

alertModalMessage.textContent = message;

alertModalOverlay.classList.add('show');

}

 

/**

* Hides the custom alert modal.

*/

function hideAlertModal() {

clickSound.play(); // Play sound on closing alert

alertModalOverlay.classList.remove('show');

}

 

/**

* Submits the checklist, compares checked items with the current level's items,

* and displays the results using the custom alert modal.

*/

function submitChecklist() {

clickSound.play(); // Play sound on submit

const levelItems = levels[currentLevel].items;

// Get all checked checkboxes and extract their data-item attribute

const checked = [...document.querySelectorAll('.custom-checkbox:checked')].map(cb => cb.dataset.item);

 

// Determine correct and wrong items

const correct = checked.filter(i => levelItems.includes(i));

// Now, 'wrong' items are those checked that are not in the 'levelItems' list.

// This implicitly handles any "wrong answers" that might have been part of the initial list

// (though the current setup only adds correct items to the list).

const wrong = checked.filter(i => !levelItems.includes(i));

 

// Build the message for the alert modal

let message = '';

if (correct.length > 0) {

message += `✅ Correct: ${correct.join(', ')}\n`;

} else {

message += `✅ Correct: None\n`;

}

if (wrong.length > 0) {

message += `❌ Wrong: ${wrong.join(', ')}\n`;

} else {

message += `❌ Wrong: None\n`;

}

 

showAlertModal(message);

}

 

// Initial load of the first level when the page loads

window.onload = function() {

loadLevel(currentLevel);

};

</script>
